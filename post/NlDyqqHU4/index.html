        registrationBean.setServlet(new HttpServlet() {

                        //
                    }
                    writer.write(String.format("cur-time : %s\n\n", new Date()));
                    writer.flush();
                }
            }
        });
        registrationBean.setUrlMappings(Collections.singletonList("/get"));
        return registrationBean;
    }
}
