public class PhoneNumberValidator implements ConstraintValidator<PhoneNumber, String> {

    @Override
    public boolean isValid(String phoneField, ConstraintValidatorContext context) {

                && (phoneField.length() > 8) && (phoneField.length() < 14);
    }
